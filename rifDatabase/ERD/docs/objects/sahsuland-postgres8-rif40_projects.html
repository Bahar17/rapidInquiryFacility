View: rif40.rif40_projects
View definition
SELECT t_rif40_projects.project, t_rif40_projects.description, t_rif40_projects.date_started, t_rif40_projects.date_ended FROM t_rif40_projects WHERE ((t_rif40_projects.project)::text IN (SELECT t_rif40_user_projects.project FROM t_rif40_user_projects WHERE (((t_rif40_user_projects.username)::name = current_user()) OR ('RIF_MANAGER'::text = (SELECT user_role_privs.granted_role FROM user_role_privs WHERE (user_role_privs.granted_role = 'RIF_MANAGER'::text)))))) ORDER BY t_rif40_projects.project;
