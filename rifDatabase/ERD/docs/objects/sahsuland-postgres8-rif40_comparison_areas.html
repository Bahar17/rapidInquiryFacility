View: rif40.rif40_comparison_areas
View definition
SELECT c.username, c.study_id, c.area_id FROM (t_rif40_comparison_areas c LEFT JOIN rif40_study_shares s ON (((c.study_id = s.study_id) AND ((s.grantee_username)::name = current_user())))) WHERE ((((c.username)::name = current_user()) OR ('RIF_MANAGER'::text = (SELECT user_role_privs.granted_role FROM user_role_privs WHERE (user_role_privs.granted_role = 'RIF_MANAGER'::text)))) OR ((s.grantee_username IS NOT NULL) AND ((s.grantee_username) IS NULL))) ORDER BY c.username;
