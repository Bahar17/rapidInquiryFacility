View: rif40.rif40_user_projects
View definition
SELECT a.project, a.username, a.grant_date, a.revoke_date, b.description, b.date_started, b.date_ended FROM t_rif40_user_projects a, t_rif40_projects b WHERE ((((a.project)::text = (b.project)::text) AND ((a.username)::name = current_user())) OR ('RIF_MANAGER'::text = (SELECT user_role_privs.granted_role FROM user_role_privs WHERE (user_role_privs.granted_role = 'RIF_MANAGER'::text)))) ORDER BY a.project;
